% Setup sectioning levels numbering options
%
% * The `tocdepth` value determines to which level the sectioning commands are
%   printed in the ToC
% * The `secnumdepth` value determines up to what level the sectioning titles
%   are numbered
%
% The sectioning levels have the following numbers:
%   -1 part     1 section     3 subsubsection  5 subparagraph
%    0 chapter  2 subsection  4 paragraph
%
\setcounter{tocdepth}{3}
\setcounter{secnumdepth}{3}

% Allow to break lines in lstlisting.
\lstset{breaklines=true}
